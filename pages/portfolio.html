void ApplyRecoil(float amount)
{
    currentRecoil += amount;
    currentRecoil = Mathf.Clamp(currentRecoil, 0f, maxRecoil);

    cameraTransform.localRotation *=
        Quaternion.Euler(-currentRecoil, 0f, 0f);
}
